set.seed(1120)

observed_fit <- penguins |>
  specify(body_mass_g ~ flipper_length_mm) |>
  fit()

boot_fits <- penguins |>
  specify(body_mass_g ~ flipper_length_mm) |>
  generate(reps = 100, type = "bootstrap") |>
  fit()

get_confidence_interval(
  boot_fits,

set.seed(1234)

null_dist <- penguins |>
  specify(body_mass_g ~ flipper_length_mm) |>
  hypothesize(null = "independence") |>
  generate(reps = 100, type = "permute") |>
  fit()
